نمونه کد

در ادامه نمونه‌ای از کد پایتون برای اسکن پورت‌های باز یک سرور را مشاهده می‌کنید:

#!/usr/bin/env python3
import socket
import sys
from datetime import datetime

# تعریف هدف
if len(sys.argv) == 2:
    target = socket.gethostbyname(sys.argv[1])  # تبدیل نام هاست به IPv4
else:
    print("تعداد آرگومان‌ها نادرست است.")
    print("نحوه استفاده: python3 port_scanner.py [هاست هدف]")
    sys.exit()

# نمایش اطلاعات اسکن
print("-" * 50)
print(f"اسکن هدف: {target}")
print(f"زمان شروع: {datetime.now()}")
print("-" * 50)

try:
    # اسکن پورت‌ها
    for port in range(1, 1025):
        s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        socket.setdefaulttimeout(1)
        result = s.connect_ex((target, port))  # بررسی اتصال
        if result == 0:
            print(f"پورت {port}: باز")
        s.close()

except KeyboardInterrupt:
    print("\nاسکن متوقف شد.")
    sys.exit()

except socket.gaierror:
    print("نام هاست قابل حل نیست.")
    sys.exit()

except socket.error:
    print("امکان اتصال به سرور وجود ندارد.")
    sys.exit()

print(f"زمان پایان: {datetime.now()}")

این کد یک اسکنر پورت ساده است که می‌تواند برای شناسایی پورت‌های باز در یک سرور استفاده شود.

نمونه کد بش

در ادامه نمونه‌ای از یک اسکریپت بش برای بررسی امنیت سیستم لینوکس را مشاهده می‌کنید:

#!/bin/bash

# اسکریپت بررسی امنیت سیستم لینوکس
echo "شروع بررسی امنیت سیستم..."
echo "-------------------------"

# بررسی کاربران با دسترسی root
echo "کاربران با دسترسی root:"
grep -Po '^sudo.+:\K.*$' /etc/group | tr ',' '\n'

# بررسی سرویس‌های در حال اجرا
echo -e "\nسرویس‌های حساس در حال اجرا:"
ps -ef | grep -E "sshd|httpd|nginx|mysql|postgresql" | grep -v grep

# بررسی پورت‌های باز
echo -e "\nپورت‌های باز:"
netstat -tuln | grep LISTEN

# بررسی آخرین لاگین‌ها
echo -e "\nآخرین لاگین‌ها:"
last | head -n 10

# بررسی فایل‌های با دسترسی نامناسب
echo -e "\nفایل‌های با دسترسی نامناسب در /etc:"
find /etc -type f -perm -o+w 2>/dev/null

echo -e "\nبررسی امنیت به پایان رسید."

نمونه کد ++C

در ادامه نمونه‌ای از کد ++C برای پیاده‌سازی یک الگوریتم رمزنگاری ساده را مشاهده می‌کنید:

#include <iostream>
#include <string>
using namespace std;

// تابع رمزنگاری با الگوریتم سزار
string encrypt(string text, int shift) {
    string result = "";
    
    // پیمایش هر کاراکتر در متن
    for (int i = 0; i < text.length(); i++) {
        // رمزنگاری حروف بزرگ انگلیسی
        if (isupper(text[i]))
            result += char(int(text[i] + shift - 65) % 26 + 65);
        
        // رمزنگاری حروف کوچک انگلیسی
        else if (islower(text[i]))
            result += char(int(text[i] + shift - 97) % 26 + 97);
        
        // کاراکترهای دیگر بدون تغییر
        else
            result += text[i];
    }
    
    return result;
}

// تابع رمزگشایی با الگوریتم سزار
string decrypt(string text, int shift) {
    return encrypt(text, 26 - shift);
}

int main() {
    string text;
    int shift;
    
    cout << "متن مورد نظر برای رمزنگاری را وارد کنید: ";
    getline(cin, text);
    
    cout << "مقدار شیفت را وارد کنید: ";
    cin >> shift;
    
    // محدود کردن شیفت به محدوده 0-25
    shift = shift % 26;
    
    string encrypted = encrypt(text, shift);
    string decrypted = decrypt(encrypted, shift);
    
    cout << "متن رمزنگاری شده: " << encrypted << endl;
    cout << "متن رمزگشایی شده: " << decrypted << endl;
    
    return 0;
}

این کد یک پیاده‌سازی ساده از الگوریتم رمزنگاری سزار است که می‌تواند برای آموزش مفاهیم اولیه رمزنگاری استفاده شود.
